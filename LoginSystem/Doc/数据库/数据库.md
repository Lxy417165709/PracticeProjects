

用户账户信息表
	用户id						-> 主键，自增
	用户手机 -> 可作为登录账号
	用户邮箱 -> 可作为登录账号
	用户密码
	用户类型	0: 普通用户
	用户注册时间
	最近登录时间
	保留字段1
	保留字段2

	
	sql:
		// 建表  这还前一些约束 
		// 还可以加个权限字段 (权限就是userType) (这里还有问题，为什么空还能写入..)
		// userType == 0 表示普通用户
		drop table tb_userAccountInformation;
		create table tb_userAccountInformation(
			userId serial primary key,
			userPhone varchar(16) unique not null,
			userEmail varchar(30) unique not null,
			userPassword varchar(20) not null,
			userType int not null,
			userRegisterTime bigint not null,
			userLastLoginTime bigint not null,
			reserved1	varchar(1),
			reserved2	varchar(1)

		)
		
		

用户个人信息表
	用户id		-> 主键，是用户账户信息表的外键
	用户头像	// 存放头像的路径
	用户名  -> 自己定的
	用户性别	
	用户联系手机
	用户联系邮箱
	用户生日
	保留字段1
	保留字段2
	
		drop table tb_userPersonalInformation;
		create table tb_userPersonalInformation(
			userId integer primary key references tb_userAccountInformation(userId),
			userPhoto varchar(80),	
			userName varchar(20),
			userSex  varchar(10),
			userContactPhone varchar(16),
			userContactEmail varchar(30),
			userBirthday bigint,
			reserved1	varchar(1),
			reserved2	varchar(1)
			
		);
